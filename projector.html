const questions = [ {}, {q: "___ Friday afternoon", a: "on"}, {q: "___ the summer", a: "in"}, {q: "___ the morning", a: "in"}, {q: "___ Tuesday", a: "on"}, {q: "___ the afternoon", a: "in"}, {q: "___ midnight", a: "at"}, {q: "___ 3 o'clock", a: "at"}, {q: "___ night", a: "at"}, {q: "___ midday", a: "at"}, {q: "___ the evening", a: "in"}, {q: "___ November", a: "in"}, {q: "___ noon", a: "at"}, {q: "___ 2nd October", a: "on"} ]; let roomPIN = ""; let currentQ = 1; let gameState = null; let timer = 15; let tId = null; let polling = null; async function api(payload) { let res = await fetch(SCRIPT_URL, { method: 'POST', body: JSON.stringify(payload) }); return await res.json(); } async function createRoom() { let res = await api({ action: 'team_create_room' }); roomPIN = res.pin; gameState = res.state; document.getElementById('setup').style.display = 'none'; document.getElementById('lobby').style.display = 'block'; document.getElementById('pin-display').innerText = roomPIN; polling = setInterval(sync, 1500); } async function sync() { let res = await api({ action: 'team_get_status', pin: roomPIN }); if(res.success) { gameState = res.state; if(gameState.status === 'LOBBY') renderLobby(); if(gameState.status === 'PLAYING') renderGameTeams(); } } function renderLobby() { let html = ''; ['A','B','C','D'].forEach(t => { let name = gameState.teams[t]; if(name) { html += `
Team ${t}: ${name}
`; } else { html += `
Team ${t}: (ว่าง)
`; } }); document.getElementById('lobby-teams').innerHTML = html; } async function kick(team) { await api({ action: 'team_update_system', pin: roomPIN, kickTeam: team }); } async function startGame() { await api({ action: 'team_update_system', pin: roomPIN, setStatus: 'PLAYING', currentQ: 1 }); document.getElementById('lobby').style.display = 'none'; document.getElementById('game').style.display = 'block'; runQuestion(); } function runQuestion() { document.getElementById('next-btn').style.display = 'none'; document.getElementById('q-num').innerText = `ข้อที่ ${currentQ} / 13`; document.getElementById('q-text').innerText = questions[currentQ].q; timer = 15; document.getElementById('timer').innerText = timer; tId = setInterval(() => { timer--; document.getElementById('timer').innerText = timer; if(timer <= 0) { clearInterval(tId); checkAnswers(); } }, 1000); } function renderGameTeams() { if(timer <= 0) return; let html = ''; ['A','B','C','D'].forEach(t => { if(!gameState.teams[t]) return; let ans = gameState.answers[t]; let status = ans ? '🔒 ล็อกแล้ว' : '⏳ คิดอยู่...'; let color = ans ? '#3b82f6' : '#475569'; html += `
Team ${t} (${gameState.scores[t]} แต้ม)
${status}
`; }); document.getElementById('game-teams').innerHTML = html; } async function checkAnswers() { await sync(); let correct = questions[currentQ].a; document.getElementById('q-text').innerHTML = `${questions[currentQ].q.replace('___', `[ ${correct} ]`)}`; let addScores = {A:0, B:0, C:0, D:0}; ['A','B','C','D'].forEach(t => { if(gameState.answers[t] === correct) addScores[t] = 1; }); await api({ action: 'team_update_system', pin: roomPIN, addScores: addScores }); let html = ''; ['A','B','C','D'].forEach(t => { if(!gameState.teams[t]) return; let bg = gameState.answers[t] === correct ? '#10b981' : '#ef4444'; html += `
Team ${t} (${gameState.scores[t] + addScores[t]} แต้ม)
ตอบ: ${gameState.answers[t] || '-'}
`; }); document.getElementById('game-teams').innerHTML = html; document.getElementById('next-btn').style.display = 'inline-block'; } async function nextQ() { currentQ++; if(currentQ > 13) { document.getElementById('q-text').innerText = "🎉 จบเกม!"; document.getElementById('next-btn').style.display = 'none'; } else { await api({ action: 'team_update_system', pin: roomPIN, currentQ: currentQ, clearAnswers: true }); runQuestion(); } }
